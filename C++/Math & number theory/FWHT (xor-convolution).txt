void FWHT(int a, ll *m, int inv) {
    for (int d = 2; d <= a; d <<= 1) {
        const int md = d >> 1;
        for (int l = 0; l < a; l += md) {
            for (int q = 0; q < md; q++, l++) {
                const ll x = m[l + md];
                m[l + md] = m[l] - x;
                m[l] += x;
            }
        }
    }
    if(inv) {
        for (int q = 0; q < a; q++) m[q] /= a;
    }
}
